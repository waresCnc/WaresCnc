<script>

let scene, camera, renderer, mesh;

function init3D() {

    scene = new THREE.Scene();

    camera = new THREE.PerspectiveCamera(75, (window.innerWidth-320)/window.innerHeight, 0.1, 1000);

    renderer = new THREE.WebGLRenderer({ antialias: true });

    document.getElementById('viewer').appendChild(renderer.domElement);

    scene.add(new THREE.AmbientLight(0xffffff, 1));

    camera.position.z = 50;

    animate();

}

function animate() { requestAnimationFrame(animate); renderer.render(scene, camera); if(mesh) mesh.rotation.z += 0.005; }

 

function procesar() {

    const file = document.getElementById('foto').files[0];

    if (!file) return alert("Sube una foto");

    const reader = new FileReader();

    reader.onload = function(e) {

        const img = new Image();

        img.src = e.target.result;

        img.onload = function() {

            const canvas = document.createElement('canvas');

            const ctx = canvas.getContext('2d');

            canvas.width = 60; canvas.height = Math.round((img.height/img.width)*60);

            ctx.drawImage(img, 0, 0, canvas.width, canvas.height);

            const data = ctx.getImageData(0, 0, canvas.width, canvas.height).data;

            

            crear3D(data, canvas.width, canvas.height);

            crearGCode(data, canvas.width, canvas.height, img.height/img.width);

        }

    };

    reader.readAsDataURL(file);

}

 

function crear3D(data, w, h) {

    if(mesh) scene.remove(mesh);

    const geo = new THREE.PlaneGeometry(w, h, w-1, h-1);

    const zM = parseFloat(document.getElementById('proZ').value);

    for(let i=0; i<geo.attributes.position.count; i++) {

        const b = (data[i*4]+data[i*4+1]+data[i*4+2])/3;

        geo.attributes.position.setZ(i, (b/255)*zM*2);

    }

    mesh = new THREE.Mesh(geo, new THREE.MeshPhongMaterial({color: 0xff6600, wireframe: true}));

    scene.add(mesh);

    camera.position.z = w;

}

 

function crearGCode(data, w, h, ratio) {

    const mmX = document.getElementById('anchoX').value;

    const mmY = (mmX * ratio).toFixed(2);

    const totalZ = parseFloat(document.getElementById('proZ').value);

    const stepZ = parseFloat(document.getElementById('stepZ').value);

    

    let g = "(WARES CNC PRO)\nG21\nG90\nG64\nM3 S15000\nG0 Z5\n";

    for(let cZ = stepZ; cZ <= totalZ + 0.01; cZ += stepZ) {

        let lim = Math.min(cZ, totalZ);

        for(let y=0; y<h; y++) {

            for(let x=0; x<w; x++) {

                let i = (y*w+x)*4;

                let b = (data[i]+data[i+1]+data[i+2])/3;

                let zV = Math.max(-((255-b)/255*totalZ), -lim).toFixed(3);

                g += `G1 X${(x*mmX/w).toFixed(2)} Y${(-(y*mmY/h)).toFixed(2)} Z${zV} F1000\n`;

            }

        }

    }

    g += "G0 Z10\nM30";

    const blob = new Blob([g], {type: 'text/plain'});

    const a = document.createElement('a');

    a.href = URL.createObjectURL(blob);

    a.download = "WARES_CNC_PROGRAMA.txt";

    document.body.appendChild(a); a.click(); document.body.removeChild(a);

}

init3D();

window.onresize = () => { renderer.setSize(window.innerWidth - (window.innerWidth > 768 ? 320 : 0), window.innerHeight); };
