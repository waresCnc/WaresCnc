<script>

    // Variables globales Three.js

    let scene, camera, renderer, mesh, controls, helperGroup;

 

    // Inicialización de la vista 3D

    function init3D() {

        const container = document.getElementById('viewer-container');

        

        scene = new THREE.Scene();

        scene.background = new THREE.Color('#050505');

        

        // Grupo para rotar fácilmente

        helperGroup = new THREE.Group();

        scene.add(helperGroup);

 

        // Cámara

        camera = new THREE.PerspectiveCamera(60, container.clientWidth / container.clientHeight, 0.1, 2000);

        camera.position.set(0, -100, 100);

 

        // Renderizador

        renderer = new THREE.WebGLRenderer({ antialias: true, alpha: false });

        renderer.setSize(container.clientWidth, container.clientHeight);

        renderer.setPixelRatio(window.devicePixelRatio);

        container.appendChild(renderer.domElement);

 

        // Controles de cámara

        controls = new THREE.OrbitControls(camera, renderer.domElement);

        controls.enableDamping = true;

        controls.dampingFactor = 0.05;

        controls.maxPolarAngle = Math.PI / 2 + 0.1; // No dejar ver mucho por debajo

        

        // Luces

        const ambientLight = new THREE.AmbientLight(0xffffff, 0.3);

        scene.add(ambientLight);

        

        const dirLight = new THREE.DirectionalLight(0xffffff, 0.8);

        dirLight.position.set(50, 50, 100);

        scene.add(dirLight);

 

        // Grid (Plano de trabajo)

        const grid = new THREE.GridHelper(200, 20, 0x444444, 0x222222);

        grid.rotation.x = Math.PI / 2; // Alinear con XY (Máquina CNC tradicional)

        scene.add(grid);

 

        // Ejes

        const axesHelper = new THREE.AxesHelper( 50 );

        scene.add( axesHelper );

 

        // Loop de renderizado

        animate();

    }

 

    function animate() { 

        requestAnimationFrame(animate); 

        controls.update(); // Necesario para damping

        renderer.render(scene, camera); 

    }

 

    // Adaptación a cambios de tamaño de ventana

    window.addEventListener('resize', () => {

        const container = document.getElementById('viewer-container');

        if(!container) return;

        camera.aspect = container.clientWidth / container.clientHeight;

        camera.updateProjectionMatrix();

        renderer.setSize(container.clientWidth, container.clientHeight);

    });

 

    // Utilidad de notificaciones

    function showToast(msg) {

        const toast = document.getElementById('toast');

        document.getElementById('toast-message').textContent = msg;

        toast.classList.add('toast-active');

        toast.classList.remove('toast-enter');

        

        setTimeout(() => {

            toast.classList.remove('toast-active');

            toast.classList.add('toast-enter');

        }, 3000);

    }

 

    // Previsualizar la imagen cargada

    function previewImage(event) {

        const file = event.target.files[0];

        if(!file) return;

        

        document.getElementById('file-name').textContent = file.name;

        document.getElementById('file-name').classList.replace('text-wares-accent', 'text-green-500');

        

        const reader = new FileReader();

        reader.onload = function(e) {

            const preview = document.getElementById('image-preview');

            preview.src = e.target.result;

            document.getElementById('image-preview-container').classList.remove('hidden');

        }

        reader.readAsDataURL(file);

    }

 

    // Función principal de procesamiento

    function procesar() {

        const file = document.getElementById('foto').files[0];

        if (!file) return showToast("⚠️ Por favor, sube una foto primero.");

        

        showToast("Procesando imagen y generando trayectoria...");

        document.getElementById('viewer-overlay').style.display = 'none';

 

        const reader = new FileReader();

        reader.onload = function(e) {

            const img = new Image();

            img.src = e.target.result;

            img.onload = function() {

                const canvas = document.createElement('canvas');

                const ctx = canvas.getContext('2d');

                

                // Resolución de procesamiento basada en el selector

                const resW = parseInt(document.getElementById('resolution').value);

                canvas.width = resW; 

                canvas.height = Math.round((img.height/img.width) * resW);

                

                // Dibujar en canvas para extraer pixeles

                ctx.drawImage(img, 0, 0, canvas.width, canvas.height);

                const data = ctx.getImageData(0, 0, canvas.width, canvas.height).data;

                

                crear3D(data, canvas.width, canvas.height);

                crearGCode(data, canvas.width, canvas.height, img.height/img.width);

            }

        };

        reader.readAsDataURL(file);

    }

 

    // Generar representación 3D de la trayectoria

    function crear3D(data, w, h) {

        if(mesh) helperGroup.remove(mesh);

        

        const mmX = parseFloat(document.getElementById('anchoX').value);

        const mmY = mmX * (h/w);

        const zM = parseFloat(document.getElementById('proZ').value);

        

        // Geometría plana con las dimensiones físicas a escala

        const geo = new THREE.PlaneGeometry(mmX, mmY, w-1, h-1);

        

        // Modificar los vértices Z basados en el color (0 negro -> fondo total, 255 blanco -> Z0)

        for(let i=0; i < geo.attributes.position.count; i++) {

            // Promedio RGB para escala de grises

            const b = (data[i*4] + data[i*4+1] + data[i*4+2]) / 3;

            // Profundidad: blanco (255) = 0, negro (0) = -zM

            const depth = -((255 - b) / 255) * zM;

            geo.attributes.position.setZ(i, depth);

        }

        

        geo.computeVertexNormals();

 

        // Material combinado (Sólido + Wireframe)

        const material = new THREE.MeshStandardMaterial({

            color: 0xff6600, 

            roughness: 0.4,

            metalness: 0.8,

            side: THREE.DoubleSide

        });

        

        mesh = new THREE.Mesh(geo, material);

        

        // Centrar la malla en el origen para que la rotación sea natural

        // En CNC el origen (0,0) suele ser una esquina, desplazamos la malla para visualización

        mesh.position.set(mmX/2, -mmY/2, 0); 

        

        helperGroup.add(mesh);

 

        // Ajustar cámara para que encuadre el objeto

        const maxDim = Math.max(mmX, mmY);

        camera.position.set(mmX/2, -mmY/2 - (maxDim*0.5), maxDim * 1.2);

        controls.target.set(mmX/2, -mmY/2, -zM/2);

    }

 

    // Generador de G-Code optimizado

    function crearGCode(data, w, h, ratio) {

        const mmX = parseFloat(document.getElementById('anchoX').value);

        const mmY = mmX * ratio;

        const totalZ = parseFloat(document.getElementById('proZ').value);

        const stepZ = parseFloat(document.getElementById('stepZ').value);

        const fRate = document.getElementById('feedRate').value;

        const sSpeed = document.getElementById('spindleSpeed').value;

        

        let g = `(WARES CNC PRO - MECANIZADO 3D)\n`;

        g += `(Dimensiones: X=${mmX.toFixed(2)}mm, Y=${mmY.toFixed(2)}mm, Profundidad=${totalZ}mm)\n`;

        g += `G21 (Milimetros)\n`;

        g += `G90 (Coordenadas Absolutas)\n`;

        g += `G64 (Modo de Velocidad Constante)\n`;

        g += `M3 S${sSpeed} (Encender Huso)\n`;

        g += `G0 Z5.000 (Subir a Z de seguridad)\n\n`;

        

        // Ciclo de pasadas de profundidad

        for(let cZ = stepZ; cZ <= totalZ + 0.01; cZ += stepZ) {

            let lim = Math.min(cZ, totalZ);

            g += `(--- PASADA Z = -${lim.toFixed(3)} ---)\n`;

            

            // Estrategia Zig-Zag para ahorrar tiempo en vaciados

            for(let y = 0; y < h; y++) {

                // Determinar dirección (0=Izq a Der, 1=Der a Izq)

                const isEven = (y % 2 === 0);

                const startX = isEven ? 0 : w - 1;

                const endX = isEven ? w : -1;

                const stepX = isEven ? 1 : -1;

 

                // Mover a Y y X inicial de la línea en rápido si es el inicio de la pasada

                if(y === 0) {

                     g += `G0 X0.00 Y0.00\n`;

                }

 

                for(let x = startX; x !== endX; x += stepX) {

                    let i = (y * w + x) * 4;

                    let b = (data[i] + data[i+1] + data[i+2]) / 3;

                    

                    // Calcular Z final del pixel. Limitado por la profundidad de pasada actual 'lim'

                    let targetZ = -((255 - b) / 255 * totalZ);

                    let zV = Math.max(targetZ, -lim).toFixed(3);

                    

                    let posX = (x * mmX / (w - 1)).toFixed(3);

                    // Y es negativo porque en la pantalla Y crece hacia abajo, 

                    // en CNC tradicional Y+ es hacia el fondo, pero mapear imágenes suele implicar Y negativo.

                    let posY = (-(y * mmY / (h - 1))).toFixed(3); 

                    

                    g += `G1 X${posX} Y${posY} Z${zV} F${fRate}\n`;

                }

            }

        }

        

        g += `\n(FIN DEL PROGRAMA)\n`;

        g += `G0 Z10.000\n`;

        g += `G0 X0.00 Y0.00\n`;

        g += `M5 (Apagar Huso)\n`;

        g += `M30 (Fin del programa y rebobinar)\n`;

        

        // Descargar Archivo

        const blob = new Blob([g], {type: 'text/plain'});

        const a = document.createElement('a');

        a.href = URL.createObjectURL(blob);

        a.download = "WARES_CNC_PROGRAMA.nc"; // Extension típica CNC (.nc, .tap o .txt)

        document.body.appendChild(a); 

        a.click(); 

        document.body.removeChild(a);

        

        showToast("✅ Archivo G-Code generado y descargado");

    }

 

    // Iniciar 3D al cargar

    window.onload = init3D;
